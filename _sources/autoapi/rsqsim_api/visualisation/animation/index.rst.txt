   :param subduction_cmap: Colourmap for subduction colorbar

   :type global_max_sub_slip: int
   :param step_size: Step size to advance every interval
   :type step_size: int
   :param interval: Time (ms) between each frame
   :type interval: int

   :type write: str
   :param fps: Frames per second.
   :type fps: int
   :param file_format: File extension for animation. Accepted values: gif, mp4, mov, avi.
   :type file_format: str
